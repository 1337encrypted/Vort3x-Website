<h1 class="center pageTitle">0x02: State Machines</h1>

<span class="note">
<h3>Example syntax to create a state machine</h3>
enum states : int8_t {
  STATE0 = 0,
  STATE1 = 1,
  STATE2 = 2
};

states status = STATE1;

<h3>Switch between states</h3>
switch(status) {
  case STATE1:
    //code
  break;

  case STATE2:
    //code
  break;

  case STATE3:
    //code
  break;

  default:
    Serial.println("CODE SHOULD NEVER ENTER THE DEFAULT CASE");
}
</span>
